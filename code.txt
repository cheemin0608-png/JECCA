        function showNearbyStops() {
            const start = document.getElementById('start').value;
            const container = document.getElementById('nearby-container');
            const list = document.getElementById('nearby-list');
            if (!start || !busStopData[start]) return;

            let html = '';
            busStopData[start].forEach(stop => {
                html += `

                </div>`;
            });
            list.innerHTML = html;
            container.style.display = 'block';
        }

        function analyzeRoute() {
            const start = document.getElementById('start').value;
            const dest = document.getElementById('destination').value;
            if (!start || !dest || start === dest) { alert("Please select valid Origin and Destination"); return; }

            document.getElementById('results').innerHTML = `<div style="text-align:center; padding:30px; font-size:12px; color:#00f2ff;">Connecting to GTFS Prediction Model...</div>`;

                renderRoutes(routes);
                stopLiveFeed();
                document.getElementById('target-label').innerText = "Select a route to view Live Feed";

                document.getElementById('val-audio').innerText = "--";
                document.getElementById('val-motion').innerText = "--";
                document.getElementById('card-motion').classList.remove('alert');
            }, 800);
        }

        function renderRoutes(routes) {
            const container = document.getElementById('results');
            let html = '';
            window.currentRouteList = routes;

            routes.forEach((r, idx) => {
                let isCrit = r.status === 'critical';
                let scoreClass = 'score-high';
                if (r.score < 90) scoreClass = 'score-med';
                if (r.score < 60) scoreClass = 'score-low';
                
                html += `
                <div class="route-card ${isCrit ? 'critical' : ''}" onclick="selectRoute(${idx}, this)">
                    <div class="route-header">
                        <div>
                            <h3 class="route-mode" style="color:${isCrit ? 'var(--danger)' : 'white'}">${r.mode}</h3>
                            <div class="route-time" style="color:${isCrit ? '#ffaaaa' : 'var(--primary)'}">${r.time}</div>
                        </div>
                        <div class="score-badge ${scoreClass}">
                            ${r.score}% Reliable
                        </div>
                    </div>
                    
                    <div style="font-size:11px; color:#888; margin-top:5px;">${r.desc}</div>
                    
                    <div class="tag-row">
                        <span class="mini-tag">${r.baseSensors.isPrivate ? 'Private' : 'Public'}</span>
                        <span class="mini-tag">GTFS Verified</span>
                    </div>

                    <div class="ai-analysis"><strong>ü§ñ AI:</strong> ${r.ai}</div>
                </div>
                `;
            });
            container.innerHTML = html;
        }

        function selectRoute(index, card) {
            document.querySelectorAll('.route-card').forEach(c => c.classList.remove('selected'));
            card.classList.add('selected');
            const routeData = window.currentRouteList[index];
            currentRouteSensors = routeData.baseSensors;
            document.getElementById('target-label').innerText = "Tracking: " + routeData.mode;
            document.getElementById('target-label').style.color = routeData.status === 'critical' ? 'var(--danger)' : 'var(--primary)';
            const motionCard = document.getElementById('card-motion');
            if (currentRouteSensors.isZero) motionCard.classList.add('alert');
            else motionCard.classList.remove('alert');
            startLiveFeed();
        }

        function startLiveFeed() {
            stopLiveFeed();
            document.getElementById('live-tag').classList.add('active');
            updateSensorValues();
            sensorInterval = setInterval(updateSensorValues, 800);
        }

        function updateSensorValues() {
            if(!currentRouteSensors) return;
            let bt;
            // Constant BT for private, Fluid for public
            if (currentRouteSensors.isPrivate) bt = currentRouteSensors.bt; 
            else bt = currentRouteSensors.bt + Math.floor(Math.random() * 10 - 5);
            if (bt < 0) bt = 0;

            let audio = currentRouteSensors.audio + Math.floor(Math.random() * 5 - 2);
            let speed;
            if (currentRouteSensors.isZero) speed = 0; 
            else {
                speed = currentRouteSensors.speed + Math.floor(Math.random() * 7 - 3);
                if (speed < 0) speed = 0;
            }
            document.getElementById('val-bt').innerText = bt;
            document.getElementById('val-audio').innerText = audio + "dB";
            document.getElementById('val-motion').innerText = speed + " km/h";
        }

        function stopLiveFeed() {
            if(sensorInterval) clearInterval(sensorInterval);
            document.getElementById('live-tag').classList.remove('active');
        }
